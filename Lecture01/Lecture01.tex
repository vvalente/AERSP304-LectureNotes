\section{Dynamics and Control}
\begin{frame}{Dynamics and Control}
    \bigskip
    \textbf{What is a model?}\par\medskip
    \begin{mynotes}
    It is a set of dynamic equations that describe the behavior of a (process/plant) system.\par
    \begin{enumerate}
      \item Physics based models: derived from first principles (e.g., Newton’s laws, conservation of mass/energy)
      \item Data driven models: derived from experimental data (e.g., system identification)
    \end{enumerate}
    \end{mynotes}

    \medskip
    \textbf{What is feedback control?}\par\medskip
    \begin{mynotes}
    Feedback control is the use of measurements of system output to influence system inputs in order to achieve desired performance.\par\smallskip

    \begin{center}
    \includegraphics[
      width=0.65\linewidth,
      alt={Thermostat feedback control block diagram showing controller, plant, and temperature feedback.}
    ]{images/thermostat.png}
    \end{center}
    \end{mynotes}
\end{frame}

\begin{frame}{Dynamics of Mechanical Systems}
    \bigskip
    Mechanical systems are composed of elements such as mass, inertia, springs, dampers, forces, and torques.

    \medskip
    \textbf{Translational Motion}\par
    \begin{itemize}
      \item Newton’s second law
      \[
        F = ma
      \]
      \item Example application: Cruise control
    \end{itemize}

    \medskip
    \textbf{Rotational Motion}\par
    \begin{itemize}
      \item Newton’s second law for rotation
      \[
        M = I\alpha
      \]
      \item Example applications: Satellite attitude control, read/write disk drives, pendulums
    \end{itemize}

    \medskip
    \textbf{Combined Translation and Rotation}\par
    An example of a system exhibiting both translational and rotational motion is a pendulum mounted on a moving cart.

\end{frame}


\begin{frame}{Dynamics of Electric Circuits}

    \bigskip
    Electric circuits are composed of elements such as resistors, capacitors, inductors, voltage and current sources, and operational amplifiers.

    \medskip
    \textbf{Circuit Analysis}\par
    \begin{itemize}
      \item Kirchhoff’s current law and Kirchhoff’s voltage law
      \item Example systems: RLC circuits, integrators
    \end{itemize}

    \medskip
    \textbf{Electromechanical Systems}\par
    Electric systems are often coupled with mechanical systems, forming electromechanical systems. An example is a DC motor, where electrical energy is converted into mechanical motion.\par
    \medskip

    \textbf{Thermal Systems}\par
    Thermal systems involve the transfer of heat within a system. Their dynamics are governed by principles such as conduction, convection, and radiation. An example is the temperature control of a heated room.

\end{frame}


\begin{frame}{Dynamics of Other Systems}


    \medskip
    \textbf{Hydraulic Systems}\par
    Hydraulic systems involve the flow and pressure of fluids within a system. Their dynamics are governed by conservation laws such as mass and energy conservation (continuity equation). An example is the water-level control of a tank system.

    \begin{mynotes}
        \medskip
        \begin{center}
        \includegraphics[
        width=0.45\linewidth,
        alt={Water tank system illustrating inflow, outflow, and liquid height dynamics}
        ]{images/tank.png}
        \end{center}

        \smallskip
        \[
        \frac{dh(t)}{dt} = \frac{1}{\rho A}\bigl(w_{in} - w_{out}\bigr),
        \qquad
        h = \frac{m}{\rho A}
        \]
    \end{mynotes}
\end{frame}

\begin{frame}{Force–Voltage Analogy for Dynamic Systems}

    \bigskip
    Mathematical equations for different physical systems can be made equivalent using analogies.

    \begin{mynotesfigures}
    \medskip
    \begin{center}
    \includegraphics[
      width=0.20\linewidth,
      alt={Mechanical mass–spring–damper system showing mass, spring, damper, and applied force}
    ]{images/mechanical.png}
    \hspace{1cm}
    \includegraphics[
      width=0.5\linewidth,
      alt={Electrical RLC circuit with voltage source, resistor, inductor, and capacitor}
    ]{images/electrical.png}
    \end{center}
    \end{mynotesfigures}

    \begin{mynotes}
    \medskip
    The mechanical system dynamics are given by:
    \[
    M\frac{d^2x(t)}{dt^2} + B\frac{dx(t)}{dt} + Kx(t) = F(t)
    \]

    \medskip
    The analogous electrical system dynamics are:
    \[
    L\frac{d^2q(t)}{dt^2} + R\frac{dq(t)}{dt} + \frac{1}{C}q(t) = V(t)
    \]
    \medskip

    \end{mynotes}

\end{frame}
